#include <stdio.h>

int main() {
    char operator;
    double num1, num2, result;
	
    printf("Enter an operator (+, -, *, /): ");
    scanf("%c", &operator);

    printf("Enter two operands: ");
    scanf("%lf %lf", &num1, &num2);

	

    switch(operator) {
	
	# For Addition 

        case '+':
            result = num1 + num2;
            printf("%.2lf + %.2lf = %.2lf", num1, num2, result);
            break;

	# For Subtraction
 
        case '-':
            result = num1 - num2;
            printf("%.2lf - %.2lf = %.2lf", num1, num2, result);
            break;

	# For Multiplication 
	
        case '*':
            result = num1 * num2;
            printf("%.2lf * %.2lf = %.2lf", num1, num2, result);
            break;

	# For Differentiation 

        case '/':
            if(num2 != 0) {
                result = num1 / num2;
                printf("%.2lf / %.2lf = %.2lf", num1, num2, result);
            } else {
                printf("Error! Division by zero.");
            }
            break;
	# IF operator does not exists

        default:
            printf("Error! Operator is not correct.");
            break;
    }

    return 0;
}
